    set [x v] to (pick random (0.0) to (1.0))
    set [y v] to (pick random (0.0) to (1.0))
    if <(((x) * (x)) + ((y) * (y))) < ((radius) * (radius)) > then // the radius is 1
        change [Count] by (1)
    end
end
